Bola.java

import java.util.Scanner;

public class Bola
{   
    //Atributos
    
    String cor;
    int tamanho;
    boolean cheia;
    String nome;
    
    //Métodos
    
    void encher()
    {
        cheia = true;
    }
    void esvaziar()
    {
        cheia = false;
    }
    void pintar(String cor)
    {
        this.cor = cor;
    }
    void definirtamanho(int tam)
    {   
        if (tam > 0)
            tamanho = tam;
    }
    void nomear(String nome)
    {
        this.nome = nome;
    }
}
